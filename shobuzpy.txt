def ip_info():
    ip = input("Enter IP or domain: ")
    print("Fetching info...")
    r = requests.get(f"https://ipinfo.io/{ip}/json")
    if r.status_code == 200:
        data = r.json()
        for key, val in data.items():
            print(f"{key}: {val}")
    else:
        print("Invalid IP or Error occurred.")
    input("\nPress Enter to return to menu...")
